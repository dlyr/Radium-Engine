444 void TimelineFrameSelector::mousePressEvent( QMouseEvent* event ) {
445  bool shiftDown = event->modifiers() & Qt::Modifier::SHIFT;
446  bool ctrlDown = event->modifiers() & Qt::Modifier::CTRL;
447  // ---------------------- LEFT CLICK --------------------------------------
448  if ( event->button() == Qt::LeftButton ) {
449  Scalar newCursor = std::max( Scalar( event->x() - m_timelineUI->m_scrollArea->getZero() ) /
450  m_timelineUI->m_scrollArea->getPixPerSec(),
451  0_ra );
452  // move cursor without render
453  if ( ctrlDown ) { onChangeCursor( newCursor, false ); }
454  // delete keyFrames between cursor and newCursor
455  else if ( shiftDown ) {
456  deleteZone( m_cursor, newCursor );
457  }
458  // move cursor and update renderer
459  else {
460  onChangeCursor( newCursor );
461  m_mouseLeftClicked = true;
462  }
463  }
464  // ------------------ RIGHT CLICK -------------------------------------
465  else if ( event->button() == Qt::RightButton ) {
466  Scalar newFrame = std::max( Scalar( event->x() - m_timelineUI->m_scrollArea->getZero() ) /
467  m_timelineUI->m_scrollArea->getPixPerSec(),
468  0_ra );
469  auto it = std::find_if( m_keyFrames.begin(), m_keyFrames.end(), [this]( const auto& t ) {
470  return Ra::Core::Math::areApproxEqual( t, m_cursor );
471  } );
472  // if already on keyFrame, move current KeyFrame
473  // ------------------- CURSOR ON KEYFRAME -----------------------
474  if ( it != m_keyFrames.end() ) {
475  Scalar nearest = nearestStep( newFrame );
476  // -------------- SINGLE MOVE -------------------------------------
477  if ( shiftDown ) {
478  // if no keyFrame under mouse, move KeyFrame to newFrame
479  auto it2 = std::find_if(
480  m_keyFrames.begin(), m_keyFrames.end(), [nearest]( const auto& t ) {
481  return Ra::Core::Math::areApproxEqual( t, nearest );
482  } );
483  if ( it2 == m_keyFrames.end() && ( std::abs( m_cursor - nearest ) > 1e-5_ra ) ) {
484  onMoveKeyFrame( m_cursor, nearest );
485  }
486  }
487  // ---------- MULTIPLE MOVE -----------------------------------
488  else {
489  // if not before preceding KeyFrame, remove or insert time
490  auto itLeft = it;
491  --itLeft;
492  Scalar left = ( it == m_keyFrames.begin() ) ? ( 0.0 ) : ( *itLeft );
493  if ( nearest > left ) { onMoveKeyFrames( m_cursor, nearest - m_cursor ); }
494  }
495  }
496  // ---------------- CURSOR NOT ON KEYFRAME --------------------------
497  else {
498  // if shiftdown, slide first right KeyFrame to the left
499  // --------------- MOVE RIGHT KEYFRAME TO THE LEFT -----------------
500  if ( shiftDown ) {
501  auto itRight = std::lower_bound( m_keyFrames.begin(), m_keyFrames.end(), newFrame );
502  // if KeyFrames on the right, remove or insert time
503  if ( itRight != m_keyFrames.end() ) {
504  onMoveKeyFrames( *itRight, newFrame - *itRight );
505  }
506  }
507  // if not shiftdown, slide first left KeyFrame to the right
508  // ---------------- MOVE LEFT KEYFRAME TO THE RIGHT -----------
509  else {
510  auto itLeft =
511  --std::lower_bound( m_keyFrames.begin(), m_keyFrames.end(), newFrame );
512  // if KeyFrames on the left, remove or insert time
513  if ( itLeft != m_keyFrames.end() ) {
514  onMoveKeyFrames( *itLeft, newFrame - *itLeft );
515  }
516  }
517  }
518  }
519  // no catch mouse event
520  else {
521  event->ignore();
522  return;
523  }
524  event->accept();
525 }

584 Scalar TimelineFrameSelector::nearestStep( Scalar time ) const {
585  Scalar deltaT =
586  TIMELINE_AUTO_SUGGEST_CURSOR_RADIUS / m_timelineUI->m_scrollArea->getPixPerSec();
587 
588  Scalar minDist = Scalar( m_timelineUI->m_durationSpin->maximum() );
589  Scalar dist;
590 
591  Scalar newCursor = time;
592 
593  for ( Scalar keyFrame : m_keyFrames ) {
594  dist = qAbs( keyFrame - time );
595  if ( dist < deltaT && dist < minDist ) {
596  minDist = dist;
597  newCursor = keyFrame;
598  }
599  if ( time < keyFrame ) break;
600  }
601 
602  Scalar step = m_timelineUI->m_scrollArea->getStep();
603  for ( int i = 0; i < m_timelineUI->m_scrollArea->getNbInterval() - 1; ++i ) {
604  Scalar pos = i * step;
605  dist = std::abs( pos - time );
606  if ( dist < deltaT && dist < minDist ) {
607  minDist = dist;
608  newCursor = pos;
609  }
610 
611  pos = i * step + 0.5_ra * step;
612  dist = std::abs( pos - time );
613  if ( dist < deltaT && dist < minDist ) {
614  minDist = dist;
615  newCursor = pos;
616  }
617 
618  if ( time < pos ) break;
619  }
620 
621  return newCursor;

625  Scalar left = std::min( time, time2 );
626  Scalar right = std::max( time, time2 );
627 
628  Scalar dist = right - left;
629 
630  // remove KeyFrames
631  auto it = m_keyFrames.begin();
632  bool first = true;
633  while ( it != m_keyFrames.end() ) {
634  Scalar keyFrame = *it;
635 
636  if ( keyFrame >= left ) {
637  it = m_keyFrames.erase( it );
638 
639  if ( keyFrame > right ) {
640  if ( first ) {
641  emit keyFramesMoved( std::distance( m_keyFrames.begin(), it ), -dist );
642  first = false;
643  }
644  it = m_keyFrames.insert( it, keyFrame - dist );
645  ++it;
646  }
647  else {
648  emit keyFrameDeleted( std::distance( m_keyFrames.begin(), it ) );
649  }
650  }
651  else {
652  ++it;
653  }
654  }
