251  const Data::ViewingParameters& viewParams,
252  const Data::ShaderProgram* shader,
253  const Data::RenderParameters& shaderParams ) {
254  if ( !m_visible || !shader ) { return; }
255  // Radium V2 : avoid this temporary
256  Core::Matrix4 modelMatrix = getTransformAsMatrix();
257  Core::Matrix4 normalMatrix = modelMatrix.inverse().transpose();
258  // bind data
259  shader->bind();
260  shader->setUniform( "transform.proj", viewParams.projMatrix );
261  shader->setUniform( "transform.view", viewParams.viewMatrix );
262  shader->setUniform( "transform.model", modelMatrix );
263  shader->setUniform( "transform.worldNormal", normalMatrix );
264  lightParams.bind( shader );
265  shaderParams.bind( shader );
266  // FIXME : find another solution for FrontFacing polygons (depends on the camera matrix)
267  // This is a hack to allow correct face culling

269  if ( viewParams.viewMatrix.determinant() < 0 ) { glFrontFace( GL_CW ); }
270  else {
271  glFrontFace( GL_CCW );
272  }
273  m_mesh->render( shader );
