178 template <typename IndexContainerType>
179 void computeIndices( Core::Geometry::LineMesh::IndexContainerType& indices,
180  IndexContainerType& other ) {
181 
182  for ( const auto& index : other ) {
183  auto s = index.size();
184  for ( unsigned int i = 0; i < s; ++i ) {
185  int i1 = index[i];
186  int i2 = index[( i + 1 ) % s];
187  if ( i1 > i2 ) std::swap( i1, i2 );
188  indices.emplace_back( i1, i2 );
189  }
190  }
191 
192  std::sort( indices.begin(),
193  indices.end(),
194  []( const Core::Geometry::LineMesh::IndexType& a,
195  const Core::Geometry::LineMesh::IndexType& b ) {
196  return a[0] < b[0] || ( a[0] == b[0] && a[1] < b[1] );
197  } );
198  indices.erase( std::unique( indices.begin(), indices.end() ), indices.end() );
199 }

694 /*
695  * Build renderTechnique for Forward Renderer : this is the default in Radium, so create Default
696  * Render Technique
697  */
698 bool ForwardRenderer::buildRenderTechnique( RenderObject* ro ) const {
699  auto material = ro->getMaterial();
700  if ( !material ) {
701  LOG( logWARNING ) << "ForwardRenderer : no material found when building RenderTechnique"
702  << " - adding red Lambertian material";
703  auto defMat = new Data::LambertianMaterial( "ForwardRenderer::Default material" );
704  defMat->m_color = Ra::Core::Utils::Color::Red();
705  material.reset( defMat );
706  }
707  auto builder = EngineRenderTechniques::getDefaultTechnique( material->getMaterialName() );
708  auto rt = Core::make_shared<RenderTechnique>();
709  // define the technique for rendering this RenderObject (here, using the default from
710  // Material name)
711  builder.second( *rt, material->isTransparent() );
712  // If renderObject is a point cloud, add geometry shader for splatting
713  auto RenderedGeometry = ro->getMesh().get();
714 
715  /*
716  * WARNING : this way of managing specific geometries is here only for testing and
717  * experimentation purpose. It will be replace soon by a better management of the way
718  * components could add specific properties to a rendertechnique
719  * TODO : see PR Draft and gist subShaderBlob
720  */
721  if ( RenderedGeometry && RenderedGeometry->getNumFaces() == 0 ) {
722  auto pointCloud = dynamic_cast<Scene::PointCloudComponent*>( ro->getComponent() );
723  if ( pointCloud ) {
724  auto addGeomShader = [&rt]( Core::Utils::Index pass ) {
725  if ( rt->hasConfiguration( pass ) ) {
726  Data::ShaderConfiguration config = rt->getConfiguration( pass );
727  config.addShader( Data::ShaderType_GEOMETRY,
728  RadiumEngine::getInstance()->getResourcesDir() +
729  "Shaders/Points/PointCloud.geom.glsl" );
730  rt->setConfiguration( config, pass );
731  }
732  };
733 
734  addGeomShader( DefaultRenderingPasses::LIGHTING_OPAQUE );
735  addGeomShader( DefaultRenderingPasses::LIGHTING_TRANSPARENT );
736  addGeomShader( DefaultRenderingPasses::Z_PREPASS );
737  // construct the parameter provider for the technique
738  auto pr = std::make_shared<PointCloudParameterProvider>( material, pointCloud );
739  rt->setParametersProvider( pr );
740  }
741  else {
742  rt->setParametersProvider( material );
743  }
744  }
745  else {
746  // make the material the parameter provider for the technique
747  rt->setParametersProvider( material );
748  }
749  ro->setRenderTechnique( rt );
750  return true;
751 }
