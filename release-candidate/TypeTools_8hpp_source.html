955 inline std::int64_t to_flag_value( std::string val ) {
956  static const std::string trueString( "true" );
957  static const std::string falseString( "false" );
958  if ( val == trueString ) { return 1; }
959  if ( val == falseString ) { return -1; }
960  val = detail::to_lower( val );
961  std::int64_t ret;
962  if ( val.size() == 1 ) {
963  if ( val[0] >= '1' && val[0] <= '9' ) {
964  return ( static_cast<std::int64_t>( val[0] ) - '0' );
965  }
966  switch ( val[0] ) {
967  case '0':
968  case 'f':
969  case 'n':
970  case '-':
971  ret = -1;
972  break;
973  case 't':
974  case 'y':
975  case '+':
976  ret = 1;
977  break;
978  default:
979  throw std::invalid_argument( "unrecognized character" );
980  }
981  return ret;
982  }
983  if ( val == trueString || val == "on" || val == "yes" || val == "enable" ) { ret = 1; }
984  else if ( val == falseString || val == "off" || val == "no" || val == "disable" ) {
985  ret = -1;
986  }
987  else {
988  ret = std::stoll( val );
989  }
