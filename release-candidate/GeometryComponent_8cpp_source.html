119 void PointCloudComponent::finalizeROFromGeometry( const Core::Asset::MaterialData* data,
120  Core::Transform transform ) {
121  // The technique for rendering this component
122  std::shared_ptr<Data::Material> roMaterial;
123  // First extract the material from asset or create a default one
124  if ( data != nullptr ) {
125  auto converter = Data::EngineMaterialConverters::getMaterialConverter( data->getType() );
126  auto mat = converter.second( data );
127  roMaterial.reset( mat );
128  }

133  Data::Mesh::getAttribName( Data::Mesh::VERTEX_COLOR ) );
134  roMaterial.reset( mat );
