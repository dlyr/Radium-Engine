46 inline std::vector<std::string> split( const std::string& s, char delim ) {
47  std::vector<std::string> elems;
48  // Check to see if empty string, give consistent result
49  if ( s.empty() ) { elems.emplace_back(); }
50  else {
51  std::stringstream ss;
52  ss.str( s );
53  std::string item;
54  while ( std::getline( ss, item, delim ) ) {
55  elems.push_back( item );
56  }
57  }
58  return elems;
59 }

63 std::string join( const T& v, std::string delim = "," ) {
64  std::ostringstream s;
65  auto beg = std::begin( v );
66  auto end = std::end( v );
67  if ( beg != end ) s << *beg++;
68  while ( beg != end ) {
69  s << delim << *beg++;
70  }
71  return s.str();
72 }

78  typename = typename std::enable_if<!std::is_constructible<std::string, Callable>::value>::type>
79 std::string join( const T& v, Callable func, std::string delim = "," ) {
80  std::ostringstream s;
81  auto beg = std::begin( v );
82  auto end = std::end( v );
83  auto loc = s.tellp();
84  while ( beg != end ) {
85  auto nloc = s.tellp();
86  if ( nloc > loc ) {
87  s << delim;
88  loc = nloc;
89  }
90  s << func( *beg++ );
91  }
92  return s.str();
93 }

279 inline void remove_default_flag_values( std::string& flags ) {
280  auto loc = flags.find_first_of( '{' );
281  while ( loc != std::string::npos ) {
282  auto finish = flags.find_first_of( "},", loc + 1 );
283  if ( ( finish != std::string::npos ) && ( flags[finish] == '}' ) ) {
284  flags.erase( flags.begin() + static_cast<std::ptrdiff_t>( loc ),
285  flags.begin() + static_cast<std::ptrdiff_t>( finish ) + 1 );
286  }
287  loc = flags.find_first_of( '{', loc + 1 );
288  }
289  flags.erase( std::remove( flags.begin(), flags.end(), '!' ), flags.end() );

343 inline std::vector<std::string> split_up( std::string str, char delimiter = '\0' ) {
344 
345  const std::string delims( "\'\"`" );
346  auto find_ws = [delimiter]( char ch ) {
347  return ( delimiter == '\0' ) ? ( std::isspace<char>( ch, std::locale() ) != 0 )
348  : ( ch == delimiter );
349  };
350  trim( str );
351 
352  std::vector<std::string> output;
353  bool embeddedQuote = false;
354  char keyChar = ' ';
355  while ( !str.empty() ) {
356  if ( delims.find_first_of( str[0] ) != std::string::npos ) {
357  keyChar = str[0];
358  auto end = str.find_first_of( keyChar, 1 );
359  while ( ( end != std::string::npos ) &&
360  ( str[end - 1] == '\\' ) ) { // deal with escaped quotes
361  end = str.find_first_of( keyChar, end + 1 );
362  embeddedQuote = true;
363  }
364  if ( end != std::string::npos ) {
365  output.push_back( str.substr( 1, end - 1 ) );
366  if ( end + 2 < str.size() ) { str = str.substr( end + 2 ); }
367  else {
368  str.clear();
369  }
370  }
371  else {
372  output.push_back( str.substr( 1 ) );
373  str = "";
374  }
375  }
376  else {
377  auto it = std::find_if( std::begin( str ), std::end( str ), find_ws );
378  if ( it != std::end( str ) ) {
379  std::string value = std::string( str.begin(), it );
380  output.push_back( value );
381  str = std::string( it + 1, str.end() );
382  }
383  else {
384  output.push_back( str );
385  str = "";
386  }
387  }
388  // transform any embedded quotes into the regular character
389  if ( embeddedQuote ) {
390  output.back() = find_and_replace(
391  output.back(), std::string( "\\" ) + keyChar, std::string( 1, keyChar ) );
392  embeddedQuote = false;
393  }
394  trim( str );
395  }
396  return output;
397 }

420 inline std::size_t escape_detect( std::string& str, std::size_t offset ) {
421  auto next = str[offset + 1];
422  if ( ( next == '\"' ) || ( next == '\'' ) || ( next == '`' ) ) {
423  auto astart = str.find_last_of( "-/ \"\'`", offset - 1 );
424  if ( astart != std::string::npos ) {
425  if ( str[astart] == ( ( str[offset] == '=' ) ? '-' : '/' ) )
426  str[offset] = ' '; // interpret this as a space so the split_up works properly
427  }
428  }
429  return offset + 1;
430 }
