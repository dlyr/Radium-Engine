609 void PolyMesh::triangulate() {
610  m_triangleIndices.clear();
611  m_triangleIndices.reserve( m_mesh.getIndices().size() );
612  for ( const auto& face : m_mesh.getIndices() ) {
613  if ( face.size() == 3 ) { m_triangleIndices.push_back( face ); }
614  else {
616  int minus { int( face.size() ) - 1 };
617  int plus { 0 };
618  while ( plus + 1 < minus ) {
619  if ( ( plus - minus ) % 2 ) {
620  m_triangleIndices.emplace_back( face[plus], face[plus + 1], face[minus] );
621  ++plus;
622  }
623  else {
624  m_triangleIndices.emplace_back( face[minus], face[plus], face[minus - 1] );
625  --minus;
626  }
627  }
628  }
629  }
