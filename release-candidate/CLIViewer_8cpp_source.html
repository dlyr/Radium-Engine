159 void CLIViewer::setCamera( Ra::Core::Utils::Index camIdx ) {
160  auto cameraManager = static_cast<Ra::Engine::Scene::CameraManager*>(
161  Ra::Engine::RadiumEngine::getInstance()->getSystem( "DefaultCameraManager" ) );
162 
163  if ( camIdx.isInvalid() || cameraManager->count() <= size_t( camIdx ) ) {
164  m_camera = new Ra::Core::Asset::Camera( m_parameters.m_size[0], m_parameters.m_size[1] );
165  m_camera->setFOV( 60.0_ra * Ra::Core::Math::toRad );
166  m_camera->setZNear( 0.1_ra );
167  m_camera->setZFar( 100_ra );
168 
169  auto aabb = Ra::Engine::RadiumEngine::getInstance()->computeSceneAabb();
170  Scalar f = m_camera->getFOV();
171  Scalar a = m_camera->getAspect();
172  const Scalar r = ( aabb.max() - aabb.min() ).norm() / 2_ra;
173  const Scalar x = r / std::sin( f / 2_ra );
174  const Scalar y = r / std::sin( f * a / 2_ra );
175  Scalar d = std::max( std::max( x, y ), 0.001_ra );
176 
177  m_camera->setPosition(
178  Ra::Core::Vector3( aabb.center().x(), aabb.center().y(), aabb.center().z() + d ) );
179  m_camera->setDirection( Ra::Core::Vector3( 0_ra, 0_ra, -1_ra ) );
180  Scalar zfar =
181  std::max( d + ( aabb.max().z() - aabb.min().z() ) * 2_ra, m_camera->getZFar() );
182  m_camera->setZFar( zfar );
183  }
184  else {
185  cameraManager->activate( camIdx );
186  m_camera = cameraManager->getActiveCamera();
187  }
188 }
