20  struct hash_vec {
21  size_t operator()( const Vector3& lvalue ) const {
22  size_t hx = std::hash<Scalar>()( lvalue[0] );
23  size_t hy = std::hash<Scalar>()( lvalue[1] );
24  size_t hz = std::hash<Scalar>()( lvalue[2] );
25  return ( hx ^ ( hy << 1 ) ) ^ hz;
26  }
27  };
28  // use a hashmap for fast search of existing vertex position
29  using VertexMap = std::unordered_map<Vector3, TopologicalMesh::VertexHandle, hash_vec>;
30  VertexMap vertexHandles;
31

34  std::vector<std::pair<AttribHandle<Vector3>, OpenMesh::HPropHandleT<Vector3>>> vprop_vec3;
35  std::vector<std::pair<AttribHandle<Vector4>, OpenMesh::HPropHandleT<Vector4>>> vprop_vec4;

39  [&triMesh, this, &vprop_float, &vprop_vec2, &vprop_vec3, &vprop_vec4]( const auto& attr ) {
40  // skip builtin attribs
41  if ( attr->getName() != std::string( "in_position" ) &&
42  attr->getName() != std::string( "in_normal" ) ) {
43  if ( attr->isFloat() )
44  addAttribPairToTopo( triMesh, attr, vprop_float, m_floatPph );

47  else if ( attr->isVector3() )

51  else
52  LOG( logWARNING )
53  << "Warning, mesh attribute " << attr->getName()
54  << " type is not supported (only float, vec2, vec3 nor vec4 are supported)";
55  }
56  } );
57 
58  size_t num_triangles = triMesh.getIndices().size();
59 
60  command.initialize( triMesh );
61 
62  const bool hasNormals = !triMesh.normals().empty();
63  if ( !hasNormals ) {
64  release_face_normals();
65  release_vertex_normals();
66  release_halfedge_normals();
67  }
68  for ( unsigned int i = 0; i < num_triangles; i++ ) {
69  std::vector<TopologicalMesh::VertexHandle> face_vhandles( 3 );
70  std::vector<TopologicalMesh::Normal> face_normals( 3 );
71  std::vector<unsigned int> face_vertexIndex( 3 );
72  const auto& triangle = triMesh.getIndices()[i];
73  for ( size_t j = 0; j < 3; ++j ) {
74  unsigned int inMeshVertexIndex = triangle[j];
75  const Vector3& p = triMesh.vertices()[inMeshVertexIndex];
76 
77  typename VertexMap::iterator vtr = vertexHandles.find( p );
78 
79  TopologicalMesh::VertexHandle vh;
80  if ( vtr == vertexHandles.end() ) {
81  vh = add_vertex( p );
82  vertexHandles.insert( vtr, typename VertexMap::value_type( p, vh ) );
83  }
84  else {
85  vh = vtr->second;
86  }
87 
88  face_vhandles[j] = vh;
89  face_vertexIndex[j] = inMeshVertexIndex;
90  if ( hasNormals ) face_normals[j] = triMesh.normals()[inMeshVertexIndex];
91  }
92 
93  TopologicalMesh::FaceHandle fh;
94 
95  // skip 2 vertex face
96  if ( face_vhandles.size() > 2 ) fh = add_face( face_vhandles );
97  // x-----------------------------------------------------------------------------------x
98 
99  if ( fh.is_valid() ) {
100  for ( size_t vindex = 0; vindex < face_vhandles.size(); vindex++ ) {
101  TopologicalMesh::HalfedgeHandle heh = halfedge_handle( face_vhandles[vindex], fh );
102  if ( hasNormals ) set_normal( heh, face_normals[vindex] );
103  property( m_inputTriangleMeshIndexPph, heh ) = face_vertexIndex[vindex];
104  copyAttribToTopo( triMesh, vprop_float, heh, face_vertexIndex[vindex] );
105  copyAttribToTopo( triMesh, vprop_vec2, heh, face_vertexIndex[vindex] );
106  copyAttribToTopo( triMesh, vprop_vec3, heh, face_vertexIndex[vindex] );
107  copyAttribToTopo( triMesh, vprop_vec4, heh, face_vertexIndex[vindex] );
108  }
109  }
110  else {
111  command.process( face_vhandles );
112  }
113  face_vhandles.clear();
114  face_normals.clear();
115  face_vertexIndex.clear();
116  }
117  command.postProcess( *this );
118 
119  // grabage collect since some wedge might already be deleted
120  garbage_collection();

337 template <typename T>
338 void TopologicalMesh::interpolatePropsOnFaces(
339  FaceHandle fh,
340  const std::vector<OpenMesh::HPropHandleT<T>>& hProps,
341  const std::vector<OpenMesh::FPropHandleT<T>>& fProps ) {
342  auto heh = halfedge_handle( fh );
343  const size_t v = valence( fh );
344 
345  // init sum to first
346  for ( size_t j = 0; j < fProps.size(); ++j ) {
347  auto hp = hProps[j];
348  auto fp = fProps[j];
349  property( fp, fh ) = property( hp, heh );
350  }
351  heh = next_halfedge_handle( heh );
352  // sum others
353  for ( size_t i = 1; i < v; ++i ) {
354  for ( size_t j = 0; j < fProps.size(); ++j ) {
355  auto hp = hProps[j];
356  auto fp = fProps[j];
357  property( fp, fh ) += property( hp, heh );
358  }
359  heh = next_halfedge_handle( heh );
360  }
361  // normalize
362  for ( size_t j = 0; j < fProps.size(); ++j ) {
363  auto fp = fProps[j];
364  property( fp, fh ) = property( fp, fh ) / v;
365  }
366 }
