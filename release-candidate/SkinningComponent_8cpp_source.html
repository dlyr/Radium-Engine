44 TriangleMesh triangulate( const PolyMesh& polyMesh ) {
45  TriangleMesh res;
46  res.setVertices( polyMesh.vertices() );
47  res.setNormals( polyMesh.normals() );
48  res.copyAllAttributes( polyMesh );
49  VectorArray<Vector3ui> indices;
50  // using the same triangulation as in Ra::Engine::PolyMesh::triangulate
51  for ( const auto& face : polyMesh.getIndices() ) {
52  if ( face.size() == 3 ) { indices.push_back( face ); }
53  else {
54  int minus { int( face.size() ) - 1 };
55  int plus { 0 };
56  while ( plus + 1 < minus ) {
57  if ( ( plus - minus ) % 2 ) {
58  indices.emplace_back( face[plus], face[plus + 1], face[minus] );
59  ++plus;
60  }
61  else {
62  indices.emplace_back( face[minus], face[plus], face[minus - 1] );
63  --minus;
64  }
65  }
66  }
67  }
68  res.setIndices( std::move( indices ) );
69  return res;
70 }

296 void SkinningComponent::createWeightMatrix() {
297  m_refData.m_bindMatrices.resize( m_refData.m_skeleton.size(), Transform::Identity() );
298  m_refData.m_weights.resize( int( m_refData.m_referenceMesh.vertices().size() ),
299  m_refData.m_skeleton.size() );
300  std::vector<Eigen::Triplet<Scalar>> triplets;
301  for ( uint col = 0; col < m_refData.m_skeleton.size(); ++col ) {
302  std::string boneName = m_refData.m_skeleton.getLabel( col );
303  auto it = m_loadedWeights.find( boneName );
304  if ( it != m_loadedWeights.end() ) {
305  const auto& W = it->second;
306  for ( uint i = 0; i < W.size(); ++i ) {
307  const auto& w = W[i];
308  int row { int( w.first ) };
309  CORE_ASSERT( row < m_refData.m_weights.rows(),
310  "Weights are incompatible with mesh." );
311  triplets.push_back( { row, int( col ), w.second } );
312  }
313  m_refData.m_bindMatrices[col] = m_loadedBindMatrices[boneName];
314  }
315  }
316  m_refData.m_weights.setFromTriplets( triplets.begin(), triplets.end() );
317 
318  checkWeightMatrix( m_refData.m_weights, false, true );
319 
320  if ( normalizeWeights( m_refData.m_weights, true ) ) {
321  LOG( logINFO ) << "Skinning weights have been normalized";
322  }

325 void SkinningComponent::computeSTBSWeights() {
326  // first skin the mesh using LBS to ensure there is no pose issue
327  // (e.g. when the mesh is in T pose but the skeleton isn't)
328  // but skip the inversion of the global mesh transform since we need the
329  // skinned vertices in model space.
330  const auto& V = m_refData.m_referenceMesh.vertices();
331  const auto& pose = m_refData.m_skeleton.getPose( SpaceType::MODEL );
332  auto vertices = Vector3Array( V.size(), Vector3::Zero() );
333  for ( int k = 0; k < m_refData.m_weights.outerSize(); ++k ) {
334  const int nonZero = m_refData.m_weights.col( k ).nonZeros();
335  WeightMatrix::InnerIterator it0( m_refData.m_weights, k );
336 #pragma omp parallel for
337  for ( int nz = 0; nz < nonZero; ++nz ) {
338  WeightMatrix::InnerIterator it = it0 + Eigen::Index( nz );
339  const uint i = it.row();
340  const uint j = it.col();
341  const Scalar w = it.value();
342  vertices[i] += w * ( pose[j] * m_refData.m_bindMatrices[j] * V[i] );
343  }
344  }
345  // compute the STBS weights
346  m_refData.m_weightSTBS = computeSTBS_weights( vertices, m_refData.m_skeleton );
347 }
