320 void KeyMappingManager::saveNode( QXmlStreamWriter& stream, const QDomNode& domNode ) {
321  if ( stream.hasError() ) { return; }
322 
323  if ( domNode.isElement() ) {
324  const QDomElement domElement = domNode.toElement();
325  if ( !domElement.isNull() ) {
326  auto tagName = domElement.tagName().toStdString();
327  stream.writeStartElement( domElement.tagName() );
328 
329  if ( tagName == "keymap" ) {
330 
331  auto saveAttrib = [&domElement, &stream]( const QString& attribName,
332  const QString& attribDefault,
333  bool optional = false ) {
334  QString attribValue = domElement.attribute( attribName, attribDefault );
335  if ( optional && attribValue == attribDefault ) { return false; }
336  stream.writeAttribute( attribName, attribValue );
337  return true;
338  };
339 
340  if ( !saveAttrib( "context", "" ) ) {
341  LOG( logERROR ) << "Error, missing context when saving keymap element";
342  return;
343  }
344  if ( !saveAttrib( "action", "" ) ) {
345  LOG( logERROR ) << "Error, missing action when saving keymap element";
346  return;
347  }
348  saveAttrib( "buttons", "" );
349  saveAttrib( "modifiers", "" );
350  saveAttrib( "key", "" );
351  saveAttrib( "wheel", "false", true );
352  }
353  else {
354  if ( domElement.hasAttributes() ) {
355  QMap<QString, QString> attributes;
356  const QDomNamedNodeMap attributeMap = domElement.attributes();
357  for ( int i = 0; i < attributeMap.count(); ++i ) {
358  const QDomNode attribute = attributeMap.item( i );
359  attributes.insert( attribute.nodeName(), attribute.nodeValue() );
360  }
361 
362  QMap<QString, QString>::const_iterator i = attributes.constBegin();
363  while ( i != attributes.constEnd() ) {
364  stream.writeAttribute( i.key(), i.value() );
365  ++i;
366  }
367  }
368  }
369 
370  if ( domElement.hasChildNodes() ) {
371  QDomNode elementChild = domElement.firstChild();
372  while ( !elementChild.isNull() ) {
373  saveNode( stream, elementChild );
374  elementChild = elementChild.nextSibling();
375  }
376  }
377 
378  stream.writeEndElement();
379  }
380  }
381  else if ( domNode.isComment() ) {
382  stream.writeComment( domNode.nodeValue() );
383  }
384  else if ( domNode.isText() ) {
385  stream.writeCharacters( domNode.nodeValue() );
386  }
