434 template <typename NonManifoldFaceCommand>
435 void TopologicalMesh::initWithWedge(
436  const Ra::Core::Geometry::MultiIndexedGeometry& mesh,
437  const Ra::Core::Geometry::MultiIndexedGeometry::LayerKeyType& layerKey,
438  NonManifoldFaceCommand command ) {
439 
440  const auto& abstractLayer = mesh.getLayer( layerKey );
441 
442  if ( !abstractLayer.hasSemantic( TriangleIndexLayer::staticSemanticName ) &&
443  !abstractLayer.hasSemantic( PolyIndexLayer::staticSemanticName ) ) {
444  LOG( logWARNING ) << "TopologicalMesh: mesh does not contains faces. Aborting conversion";
445  return;
446  }
447 
448  clean();
449 
450  LOG( logINFO ) << "TopologicalMesh: load mesh with "
451  //<< abstractLayer.size()
452  << " faces and " << mesh.vertices().size() << " vertices.";
453  // use a hashmap for fast search of existing vertex position
454  using VertexMap = std::unordered_map<Vector3, TopologicalMesh::VertexHandle, hash_vec>;
455  VertexMap vertexHandles;
456 
457  // loop over all attribs and build correspondance pair
458  mesh.vertexAttribs().for_each_attrib( InitWedgeAttribsFromMultiIndexedGeometry { this, mesh } );
459 
460  for ( size_t i = 0; i < mesh.vertices().size(); ++i ) {
461  // create an empty wedge, with 0 ref
462  Wedge w;
463 
464  WedgeData wd;
465  wd.m_position = mesh.vertices()[i];
466  copyMeshToWedgeData( mesh,
467  i,
468  m_wedges.m_wedgeFloatAttribHandles,
469  m_wedges.m_wedgeVector2AttribHandles,
470  m_wedges.m_wedgeVector3AttribHandles,
471  m_wedges.m_wedgeVector4AttribHandles,
472  &wd );
473  // here ref is not incremented
474  w.setWedgeData( std::move( wd ) );
475  // the newly added wedge is not referenced yet, will be done with `newReference` when
476  // creating faces just below
477  m_wedges.m_data.push_back( w );
478  }
479 
480  LOG( logINFO ) << "TopologicalMesh: have " << m_wedges.size() << " wedges ";
481 
482  const bool hasNormals = !mesh.normals().empty();
483  if ( !hasNormals ) {
484  release_face_normals();
485  release_vertex_normals();
486  release_halfedge_normals();
487  }
488 
489  command.initialize( mesh );
490 
491  auto processFaces = [&mesh, &vertexHandles, this, hasNormals, &command]( const auto& faces ) {
492  size_t num_triangles = faces.size();
493  for ( unsigned int i = 0; i < num_triangles; i++ ) {
494  const auto& face = faces[i];
495  const size_t num_vert = face.size();
496  std::vector<TopologicalMesh::VertexHandle> face_vhandles( num_vert );
497  std::vector<TopologicalMesh::Normal> face_normals( num_vert );
498  std::vector<WedgeIndex> face_wedges( num_vert );
499 
500  for ( size_t j = 0; j < num_vert; ++j ) {
501  unsigned int inMeshVertexIndex = face[j];
502  const Vector3& p = mesh.vertices()[inMeshVertexIndex];
503 
504  typename VertexMap::iterator vtr = vertexHandles.find( p );
505  TopologicalMesh::VertexHandle vh;
506  if ( vtr == vertexHandles.end() ) {
507  vh = add_vertex( p );
508  vertexHandles.insert( vtr, typename VertexMap::value_type( p, vh ) );
509  }
510  else {
511  vh = vtr->second;
512  }
513 
514  face_vhandles[j] = vh;
515  if ( hasNormals ) face_normals[j] = mesh.normals()[inMeshVertexIndex];
516  face_wedges[j] = WedgeIndex { inMeshVertexIndex };
517  m_wedges.m_data[inMeshVertexIndex].getWedgeData().m_vertexHandle = vh;
518  }
519 
520  // remove consecutive equal vertex
521  // first take care of "loop" if begin == *end-1
522  // apply the same modifications on wedges and normals
523  // e.g. 1 2 1 becomes 1 2
524  {
525  auto begin = face_vhandles.begin();
526  if ( face_vhandles.size() > 2 ) {
527  auto end = face_vhandles.end() - 1;
528  auto wedgeEnd = face_wedges.end() - 1;
529  auto normalEnd = face_normals.end() - 1;
530 
531  while ( begin != end && *begin == *end ) {
532  end--;
533  wedgeEnd--;
534  normalEnd--;
535  }
536  face_vhandles.erase( end + 1, face_vhandles.end() );
537  face_wedges.erase( wedgeEnd + 1, face_wedges.end() );
538  face_normals.erase( normalEnd + 1, face_normals.end() );
539  }
540  }
541  // then remove duplicates
542  // e.g. 1 2 2 becomes 1 2
543  // equiv of
544  // face_vhandles.erase( std::unique( face_vhandles.begin(), face_vhandles.end() ),
545  // face_vhandles.end() );
546  // but handles wedges and normals
547  // see (https://en.cppreference.com/w/cpp/algorithm/unique)
548  {
549  auto first = face_vhandles.begin();
550  auto wedgeFirst = face_wedges.begin();
551  auto normalFirst = face_normals.begin();
552  auto last = face_vhandles.end();
553 
554  if ( first != last ) {
555  auto result = first;
556  auto wedgeResult = wedgeFirst;
557  auto normalResult = normalFirst;
558  while ( ++first != last ) {
559  if ( !( *result == *first ) ) {
560  ++result;
561  ++wedgeResult;
562  ++normalResult;
563  if ( result != first ) {
564  *result = std::move( *first );
565  *wedgeResult = std::move( *wedgeFirst );
566  *normalResult = std::move( *normalFirst );
567  }
568  }
569  }
570  face_vhandles.erase( result + 1, face_vhandles.end() );
571  face_wedges.erase( wedgeResult + 1, face_wedges.end() );
572  face_normals.erase( normalResult + 1, face_normals.end() );
573  }
574  }
575 
577  // unique sort size == vhandles size, if not split ...
578 
579  TopologicalMesh::FaceHandle fh;
580  // skip 2 vertex face
581  if ( face_vhandles.size() > 2 ) fh = add_face( face_vhandles );
582 
583  // In case of topological inconsistancy, face will be invalid (or uninitialized <>
584  // invalid)
585  if ( fh.is_valid() ) {
586  for ( size_t vindex = 0; vindex < face_vhandles.size(); vindex++ ) {
587  TopologicalMesh::HalfedgeHandle heh =
588  halfedge_handle( face_vhandles[vindex], fh );
589  if ( hasNormals ) set_normal( heh, face_normals[vindex] );
590  property( m_wedgeIndexPph, heh ) = m_wedges.newReference( face_wedges[vindex] );
591  }
592  }
593  else {
594  command.process( face_vhandles );
595  }
596  face_vhandles.clear();
597  face_normals.clear();
598  }
599  };

601  if ( abstractLayer.hasSemantic( TriangleIndexLayer::staticSemanticName ) ) {
602  const auto& faces = static_cast<const TriangleIndexLayer&>( abstractLayer ).collection();
603  LOG( logINFO ) << "TopologicalMesh: process " << faces.size() << " triangular faces ";
604  processFaces( faces );
605  }
606  else if ( abstractLayer.hasSemantic( PolyIndexLayer::staticSemanticName ) ) {
607  const auto& faces = static_cast<const PolyIndexLayer&>( abstractLayer ).collection();
608  LOG( logINFO ) << "TopologicalMesh: process " << faces.size() << " polygonal faces ";
609  processFaces( faces );
610  }
611 
612  command.postProcess( *this );
613  if ( hasNormals ) {
614  m_normalsIndex = m_wedges.getWedgeAttribIndex<Normal>( "in_normal" );
615 
616  m_vertexFaceWedgesWithSameNormals.clear();
617  m_vertexFaceWedgesWithSameNormals.resize( n_vertices() );
618 
619  for ( auto itr = vertices_begin(), stop = vertices_end(); itr != stop; ++itr ) {
620  std::unordered_map<TopologicalMesh::Normal,
621  std::pair<std::set<FaceHandle>, std::set<WedgeIndex>>,
622  hash_vec>
623  normalSharedByWedges;
624 
625  auto vh = *itr;
626 
627  for ( ConstVertexIHalfedgeIter vh_it = cvih_iter( vh ); vh_it.is_valid(); ++vh_it ) {
628  const auto& widx = property( m_wedgeIndexPph, *vh_it );
629  if ( widx.isValid() && !m_wedges.getWedge( widx ).isDeleted() ) {

631  normalSharedByWedges[oldNormal].first.insert( face_handle( *vh_it ) );
632  normalSharedByWedges[oldNormal].second.insert( widx );
633  }
634  }
635 
636  for ( const auto& pair : normalSharedByWedges ) {
637  for ( const auto& fh : pair.second.first ) {
638  auto& v = m_vertexFaceWedgesWithSameNormals[vh.idx()][fh.idx()];
639  v.insert( v.end(), pair.second.second.begin(), pair.second.second.end() );
640  }
641  }
642  }
643  }
644  LOG( logINFO ) << "TopologicalMesh: load end with " << m_wedges.size() << " wedges ";
645 }
