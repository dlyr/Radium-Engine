418  globjects::init( getProcAddress );
419  // mark openGL as initialized
420  m_glInitialized = true;
421 
422  LOG( logINFO ) << "*** Radium Engine OpenGL context ***";
423  LOG( logINFO ) << "Renderer (glbinding) : " << glbinding::aux::ContextInfo::renderer();
424  LOG( logINFO ) << "Vendor (glbinding) : " << glbinding::aux::ContextInfo::vendor();
425  LOG( logINFO ) << "OpenGL (glbinding) : "
426  << glbinding::aux::ContextInfo::version().toString();
427  LOG( logINFO ) << "GLSL : "
428  << gl::glGetString( gl::GLenum( GL_SHADING_LANGUAGE_VERSION ) );
429 
430  LOG( logINFO ) << "*** Radium Engine Viewer ***";
431 
432  // emit the signal so that the client will initialize the OpenGL part of the Engine
433  // and custom OpenGL properties
434  emit requestEngineOpenGLInitialization();
435 
436  // Configure the viewer services
437  // create default camera interface : trackball
438  m_camera = std::make_unique<TrackballCameraManipulator>();
439  m_camera->getCamera()->setViewport( width(), height() );
440  auto headlight = new Engine::Scene::DirectionalLight(
441  Ra::Engine::Scene::SystemEntity::getInstance(), "headlight" );
442  headlight->setColor( Ra::Core::Utils::Color::Grey( 1.0_ra ) );
443 
444  m_camera->attachLight( headlight );
445 
446  // Register to the camera manager active camera changes
447  auto cameraManager = static_cast<Ra::Engine::Scene::CameraManager*>(
448  Engine::RadiumEngine::getInstance()->getSystem( "DefaultCameraManager" ) );
449  cameraManager->activeCameraObservers().attach( [this]( Core::Utils::Index /*idx*/ ) {
450  m_camera->updateCamera();
451  m_camera->getCamera()->setViewport( width(), height() );
452  } );
453 
454  // Initialize renderers added to the viewer before initializeGL
455  for ( auto& rptr : m_pendingRenderers ) {
456  initializeRenderer( rptr.get() );
457  }
458  m_pendingRenderers.clear();
459 
460  // create the gizmo manager (Ui)
461  createGizmoManager();
462 
463  // Signal that OpenGL is initialized
464  emit glInitialized();
465 
466  // If no renderer was added before that (either by slots on requestEngineOpenGLInitialization
467  // or on glInitialized), add default forward renderer
468  if ( m_renderers.empty() ) {
469  LOG( logINFO ) << "[Viewer] No renderer added, adding default (Forward Renderer)";
470  std::shared_ptr<Ra::Engine::Rendering::Renderer> e(
471  new Ra::Engine::Rendering::ForwardRenderer() );
472  addRenderer( e );
473  }
474 
475  if ( m_currentRenderer == nullptr ) { changeRenderer( 0 ); }
476 
477  return m_glInitialized;
478 }

608 bool Viewer::handleKeyPressEvent( QKeyEvent* event ) {
609  bool eventCatched = false;
610 
611  auto [actionCamera, actionGizmo, actionViewer] =
612  getComponentActions( Qt::NoButton, event->modifiers(), activeKey(), false );
613 
614  // Is keymapping something of the viewer only ?
615  // or should be dispatched to all receivers ?
616 
617  if ( actionCamera.isValid() ) {
618  eventCatched = m_camera->handleKeyPressEvent( event, actionCamera );
619  }
620  else if ( actionGizmo.isValid() ) {
621  // m_gizmoManager->handleKeyPressEvent( event, action );
622  // eventCatched = true;
623  }
624  else if ( actionViewer.isValid() ) {
625 
626  if ( actionViewer == VIEWER_TOGGLE_WIREFRAME ) {
627  m_currentRenderer->toggleWireframe();
628  eventCatched = true;
629  }
630  else if ( actionViewer == VIEWER_RELOAD_SHADERS ) {
631  reloadShaders();
632  eventCatched = true;
633  }
634  else if ( actionViewer == VIEWER_PICKING_MULTI_CIRCLE ) {
635  m_isBrushPickingEnabled = !m_isBrushPickingEnabled;
636  m_currentRenderer->setBrushRadius( m_isBrushPickingEnabled ? m_brushRadius : 0 );
637  emit toggleBrushPicking( m_isBrushPickingEnabled );
638  eventCatched = true;
639  }
640  else if ( actionViewer == VIEWER_SWITCH_CAMERA ) {
641  auto cameraManager = static_cast<Ra::Engine::Scene::CameraManager*>(
642  Engine::RadiumEngine::getInstance()->getSystem( "DefaultCameraManager" ) );
643  static int idx = 0;
644  if ( cameraManager->count() > 0 ) {
645  idx %= cameraManager->count();
646  cameraManager->activate( idx );
647  eventCatched = true;
648  }
649  idx++;
650  }
651  else if ( actionViewer == VIEWER_CAMERA_FIT_SCENE ) {
652  fitCamera();
653  eventCatched = true;
654  }
655  else if ( actionViewer == VIEWER_HELP ) {
656  displayHelpDialog();
657  eventCatched = true;
658  requestActivate();
659  }
660  else {
661  auto itr = m_customKeyActions[KeyEventType::KeyPressed].find( actionViewer );
662  if ( itr != m_customKeyActions[KeyEventType::KeyPressed].end() ) {
663  itr->second( event );
664  eventCatched = true;
665  }
666  }
667  }
668 
669  return eventCatched;
670 }
